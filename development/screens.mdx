## Overview

Casa de Historias follows a feature-based screen organization with a centralized navigation system. All screens use Jetpack Compose and follow MVVM architecture with ViewModels for state management.

## Navigation System

### Screen Definitions

All screens are defined as sealed class objects in `NavGraph.kt`:

<CodeGroup>
```kotlin NavGraph.kt
sealed class Screen(val route: String) {
    object Welcome : Screen("welcome")
    object Discover : Screen("discover")
    object Dictionary : Screen("dictionary")
    object Login : Screen("login")
    object SignUp : Screen("signup")
    object StoryList : Screen("story_list")
    object AddStory : Screen("add_story")
    object StoryMap : Screen("story_map")
    object Favorites : Screen("favorites")
    object Profile : Screen("profile")
    object Search : Screen("search")
    object Settings : Screen("settings")
    
    // Screens with parameters
    object StoryDetail : Screen("story_detail/{storyId}") {
        fun createRoute(storyId: String) = "story_detail/$storyId"
    }
    object EditStory : Screen("edit_story/{storyId}") {
        fun createRoute(storyId: String) = "edit_story/$storyId"
    }
}
```
</CodeGroup>

### Navigation Graph

The `NavGraph` composable sets up all navigation routes:

<CodeGroup>
```kotlin NavGraph Setup
@Composable
fun NavGraph(
    navController: NavHostController,
    authViewModel: AuthViewModel,
    startDestination: String
) {
    val storyViewModel: StoryViewModel = hiltViewModel()
    val favoritesViewModel: FavoritesViewModel = hiltViewModel()

    NavHost(
        navController = navController,
        startDestination = startDestination
    ) {
        composable(Screen.Welcome.route) {
            WelcomeScreen(
                onContinue = {
                    navController.navigate(Screen.Login.route) {
                        popUpTo(Screen.Welcome.route) { inclusive = true }
                    }
                }
            )
        }
        
        composable(Screen.StoryList.route) {
            StoryListScreen(
                viewModel = storyViewModel,
                onStoryClick = { storyId ->
                    navController.navigate(
                        Screen.StoryDetail.createRoute(storyId)
                    )
                },
                onAddStoryClick = {
                    navController.navigate(Screen.AddStory.route)
                }
            )
        }
        
        // Parameterized route
        composable(
            route = Screen.StoryDetail.route,
            arguments = listOf(
                navArgument("storyId") { type = NavType.StringType }
            )
        ) { backStackEntry ->
            val storyId = backStackEntry.arguments?.getString("storyId") ?: ""
            // Screen implementation
        }
    }
}
```
</CodeGroup>

## Screen Patterns

### Basic Screen Template

Most screens follow this structure:

<CodeGroup>
```kotlin Screen Template
@OptIn(ExperimentalMaterial3Api::class)
@Composable
fun MyScreen(
    viewModel: MyViewModel = hiltViewModel(),
    onNavigateBack: () -> Unit = {},
    onNavigateToDetail: (String) -> Unit = {}
) {
    // Collect state from ViewModel
    val uiState by viewModel.uiState.collectAsState()
    val items by viewModel.items.collectAsState()
    
    Scaffold(
        topBar = {
            TopAppBar(
                title = { Text("My Screen") },
                navigationIcon = {
                    IconButton(onClick = onNavigateBack) {
                        Icon(Icons.AutoMirrored.Filled.ArrowBack, 
                            contentDescription = "Back")
                    }
                },
                colors = TopAppBarDefaults.topAppBarColors(
                    containerColor = NocheProfunda
                )
            )
        },
        containerColor = NocheProfunda
    ) { padding ->
        // Content based on state
        when (uiState) {
            is UiState.Loading -> LoadingView()
            is UiState.Error -> ErrorView()
            is UiState.Success -> ContentView(items)
        }
    }
}
```
</CodeGroup>

Used in `StoryListScreen.kt` and `FavoritesScreen.kt`:

<CodeGroup>
```kotlin List Screen
@Composable
fun StoryListScreen(
    viewModel: StoryViewModel,
    onStoryClick: (String) -> Unit = {},
    onAddStoryClick: () -> Unit = {}
) {
    val stories by viewModel.stories.collectAsState()
    val isLoading by viewModel.isLoading.collectAsState()
    val errorMessage by viewModel.errorMessage.collectAsState()

    Scaffold(
        topBar = { /* ... */ },
        floatingActionButton = {
            FloatingActionButton(
                onClick = onAddStoryClick,
                containerColor = AguaSagrada
            ) {
                Icon(Icons.Default.Add, contentDescription = "Add")
            }
        }
    ) { padding ->
        when {
            isLoading -> LoadingState()
            errorMessage != null -> ErrorState(errorMessage)
            stories.isEmpty() -> EmptyState()
            else -> {
                LazyColumn(
                    modifier = Modifier.padding(padding),
                    contentPadding = PaddingValues(20.dp),
                    verticalArrangement = Arrangement.spacedBy(16.dp)
                ) {
                    items(stories, key = { it.id }) { story ->
                        ModernStoryCard(
                            story = story,
                            onClick = { onStoryClick(story.id) }
                        )
                    }
                }
            }
        }
    }
}
```

## Best Practices

<AccordionGroup>
  <Accordion title="Use callback parameters for navigation">
    Pass navigation callbacks as parameters instead of accessing NavController directly in screens:
    ```kotlin
    @Composable
    fun MyScreen(
        onNavigateToDetail: (String) -> Unit,
        onNavigateBack: () -> Unit
    )
    ```
  </Accordion>

  <Accordion title="Handle navigation arguments safely">
    Always provide defaults when extracting navigation arguments:
    ```kotlin
    val storyId = backStackEntry.arguments?.getString("storyId") ?: ""
    ```
  </Accordion>

  <Accordion title="Clear backstack for auth flows">
    Use `popUpTo(0) { inclusive = true }` when navigating after login/logout to clear the entire backstack.
  </Accordion>

  <Accordion title="Hoist state appropriately">
    Keep UI-only state in composables with `remember`, but use ViewModels for business logic and data that survives configuration changes.
  </Accordion>

  <Accordion title="Use sealed classes for navigation">
    Define all routes in a sealed class hierarchy to prevent typos and enable type-safe navigation.
  </Accordion>
</AccordionGroup>
