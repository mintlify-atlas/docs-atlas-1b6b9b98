## Overview

Casa de Historias uses Jetpack Compose for building modern, declarative UI. This guide covers common UI patterns, custom components, and design conventions used across the app.

## Custom Components

### Story Card

A modern card component with gradient overlays and image support:

<CodeGroup>
```kotlin ModernStoryCard
@OptIn(ExperimentalMaterial3Api::class)
@Composable
fun ModernStoryCard(
    story: Story,
    onClick: () -> Unit = {}
) {
    Card(
        modifier = Modifier
            .fillMaxWidth()
            .height(200.dp),
        shape = RoundedCornerShape(24.dp),
        border = BorderStroke(1.dp, Color.White.copy(alpha = 0.1f)),
        elevation = CardDefaults.cardElevation(
            defaultElevation = 8.dp,
            pressedElevation = 12.dp
        ),
        colors = CardDefaults.cardColors(
            containerColor = FondoTarjeta
        ),
        onClick = onClick
    ) {
        Box(modifier = Modifier.fillMaxSize()) {
            // Background image with gradient overlay
            if (!story.imageUrl.isNullOrBlank()) {
                AsyncImage(
                    model = story.imageUrl,
                    contentDescription = null,
                    modifier = Modifier.fillMaxSize(),
                    contentScale = ContentScale.Crop
                )
                Box(
                    modifier = Modifier
                        .fillMaxSize()
                        .background(
                            Brush.verticalGradient(
                                colors = listOf(
                                    Color.Transparent,
                                    NocheProfunda.copy(alpha = 0.6f),
                                    NocheProfunda.copy(alpha = 0.95f)
                                )
                            )
                        )
                )
            }
            
            // Content layer
            Column(
                modifier = Modifier
                    .fillMaxSize()
                    .padding(20.dp),
                verticalArrangement = Arrangement.Bottom
            ) {
                // Nahuatl title
                if (story.titleNahuatl.isNotBlank()) {
                    Text(
                        text = story.titleNahuatl,
                        style = MaterialTheme.typography.titleMedium,
                        fontStyle = FontStyle.Italic,
                        color = AguaSagrada
                    )
                }
                // Spanish title
                Text(
                    text = story.titleEs,
                    style = MaterialTheme.typography.titleLarge,
                    fontWeight = FontWeight.Bold,
                    color = LunaLlena
                )
            }
        }
    }
}
```
</CodeGroup>

### Custom Text Field

Stylized `OutlinedTextField` with consistent theming:

<CodeGroup>
```kotlin CustomOutlinedTextField
@Composable
fun CustomOutlinedTextField(
    value: String,
    onValueChange: (String) -> Unit,

    icon: ImageVector,

    singleLine: Boolean = true,

        onValueChange = onValueChange,

            unfocusedLabelColor = TextoSecundario,
            cursorColor = AguaSagrada,
            focusedTextColor = LunaLlena,
            unfocusedTextColor = LunaLlena
        )
    )
}

<CodeGroup>

val isLoading by viewModel.isLoading.collectAsState()
val errorMessage by viewModel.errorMessage.collectAsState()

when {
    isLoading -> {
        Box(
            modifier = Modifier.fillMaxSize(),
            contentAlignment = Alignment.Center
        ) {
            Column(horizontalAlignment = Alignment.CenterHorizontally) {
                CircularProgressIndicator(color = AguaSagrada)
                Text(
                    "Cargando historias...",
                    color = TextoSecundario
                )
            }
        }
    }
    errorMessage != null -> {
        // Error state
    }
    stories.isEmpty() -> {
        // Empty state
    }
    else -> {
        // Content state
    }
}
```
</CodeGroup>

var showTitle by remember { mutableStateOf(false) }
var showSubtitle by remember { mutableStateOf(false) }

    showSubtitle = true

## Best Practices

<AccordionGroup>
  <Accordion title="Use state hoisting">
    Lift state to parent composables and pass callbacks down for better reusability:
    ```kotlin
    @Composable
    fun StoryCard(
        story: Story,
        onClick: () -> Unit  // Callback instead of internal navigation
    )
    ```
  </Accordion>

  <Accordion title="Apply consistent spacing">
    Use standardized spacing values:
    - Cards: `16.dp` or `20.dp` between items
    - Padding: `16.dp` or `20.dp` horizontal/vertical
    - Corner radius: `12.dp`, `16.dp`, or `24.dp`
  </Accordion>

  <Accordion title="Use remember for non-observable state">
    Use `remember` for UI-only state and `collectAsState()` for ViewModel flows:
    ```kotlin
    var showDialog by remember { mutableStateOf(false) }
    val stories by viewModel.stories.collectAsState()
    ```
  </Accordion>

  <Accordion title="Provide content descriptions">
    Always include content descriptions for accessibility:
    ```kotlin
    Icon(
        Icons.Default.Add,
        contentDescription = "Añadir Historia"
    )
    ```
  </Accordion>
</AccordionGroup>
